Agent的关键组件：

  • 大语言模型（LLM）：作为"大脑"，负责推理和决策
  • 工具调用（Tool Use）：搜索、计算器、代码执行、API调用
  • 记忆系统：短期记忆（上下文窗口）+ 长期记忆（向量数据库）
  • 规划能力：任务分解、反思、自我纠错
# 简化的AI Agent伪代码
class AIAgent:
    def __init__(self, llm, tools):
        self.llm = llm          # 大语言模型
        self.tools = tools      # 可用工具列表
        self.memory = []        # 对话历史

    def think(self, task):
        """规划：分解任务"""
        plan = self.llm.generate(
            f"任务: {task}\n"
            f"请制定执行步骤:\n"
            f"1. ..."
        )
        return plan

    def act(self, step):
        """行动：执行步骤"""
        if step.startswith("调用工具:"):
            tool_name = step.split(":")[1]
            result = self.tools[tool_name].run()
            self.memory.append({"role": "tool", "content": result})
            return result
        elif step.startswith("生成回复:"):
            return self.llm.generate(step)
        else:
            return self.execute_code(step)

    def run(self, task):
        """完整的执行循环"""
        plan = self.think(task)
        results = []
        for step in plan:
            result = self.act(step)
            results.append(result)
            if not self.verify(result, task):
                plan = self.revise_plan(plan, result)
        return results
